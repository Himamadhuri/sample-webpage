function validate(){  
    let emailid=document.getElementById("email").value;
    let org=document.getElementById("orgname").value;
    let fullname=document.getElementById("fullname").value;
    if(fullname!="" && emailid!="" && org!="")
    {
    alert("Form Submitted"); 
    document.getElementById("details").reset();
    document.getElementById("err").innerHTML="";
    }    
    if(fullname==""){
        document.getElementById("err").innerHTML="please fill all the required fields below";
        document.getElementById("name-alert").innerText="*Name is required";
    }
    else 
    document.getElementById("name-alert").innerHTML="*";
    
    if(emailid==""){
        document.getElementById("err").innerHTML="please fill all the required fields below";
        document.getElementById("email-alert").innerText="*Email is required";
    }
    else
    document.getElementById("email-alert").innerHTML="*";
    if(org==""){
        document.getElementById("err").innerHTML="please fill all the required fields below";
        document.getElementById("orgno-alert").innerHTML="*Organization Name is required";
    }
    else
    document.getElementById("orgno-alert").innerHTML="*";
    return false;   
}
function name_validate(){    
    let fullname=document.getElementById("fullname").value;    
    if(fullname==""){       
        document.getElementById("name-alert").innerText="*Name is required";
    }
    else if(fullname!="")
    document.getElementById("name-alert").innerHTML="*";
}
function email_validate(){
    let emailid=document.getElementById("email").value;
    if(emailid==""){       
        document.getElementById("email-alert").innerText="*Email is required";
    }
    else if(emailid!="")
    document.getElementById("email-alert").innerHTML="*";
    let mailformat = /[a-zA-Z0-9\.]+@[a-zA-Z]+\.[a-z\.]+/;
    if (!emailid.match(mailformat)) 
    document.getElementById("email-alert").innerText="*Invalid Email";  
}
